```javascript
export const handleError = (error) => {
  const errorMap = {
    400: '请求参数错误',
    401: '未授权，请登录',
    403: '拒绝访问',
    404: '请求的资源不存在',
    500: '服务器内部错误',
    502: '网关错误',
    503: '服务暂时不可用'
  }
  
  let message = '操作失败'
  if (error.response) {
    message = error.response.data?.message || errorMap[error.response.status] || message
  } else if (error.request) {
    message = '网络连接失败，请检查网络'
  }
  
  ElMessage.error(message)
  return message
}
```

#### 3.1 SEO 优化

**目标**：提升搜索引擎收录效果

**实施要点**：

1. **动态 Meta 标签**
```javascript
// blog-web/src/utils/seo.js
export const updateMetaTags = (article) => {
  document.title = `${article.title} - 我的博客`
  
  const metaTags = [
    { name: 'description', content: article.summary },
    { name: 'keywords', content: article.tags.map(t => t.name).join(',') },
    { property: 'og:title', content: article.title },
    { property: 'og:description', content: article.summary },
    { property: 'og:image', content: article.cover }
  ]
  
  metaTags.forEach(({ name, property, content }) => {
    let meta = document.querySelector(
      `meta[${name ? 'name' : 'property'}="${name || property}"]`
    )
    if (!meta) {
      meta = document.createElement('meta')
      if (name) meta.name = name
      if (property) meta.setAttribute('property', property)
      document.head.appendChild(meta)
    }
    meta.content = content
  })
}
```

2. **结构化数据（JSON-LD）**
```javascript
export const addStructuredData = (article) => {
  const schema = {
    "@context": "https://schema.org",
    "@type": "BlogPosting",
    "headline": article.title,
    "image": article.cover,
    "datePublished": article.created_at,
    "description": article.summary
  }
  
  let script = document.querySelector('script[type="application/ld+json"]')
  if (!script) {
    script = document.createElement('script')
    script.type = 'application/ld+json'
    document.head.appendChild(script)
  }
  script.textContent = JSON.stringify(schema)
}
```

```go
func (h *Handler) GenerateRSS(c *gin.Context) {
    articles, _, err := h.articleRepo.ListPublished(0, 20, 0, 0, "")
    if err != nil {
        middleware.Error(c, err)

            Content:     article.Content,
            Created:     article.CreatedAt,
        })
    }
    
    rss, _ := feed.ToRss()
    c.Header("Content-Type", "application/xml")
    c.String(200, rss)
}
```

```go
// internal/api/v1/sitemap/handler.go
func (h *Handler) GenerateSitemap(c *gin.Context) {
    articles, _, _ := h.articleRepo.ListPublished(0, 1000, 0, 0, "")
    
    sitemap := `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">`
    
    // 首页
    sitemap += fmt.Sprintf(`
  <url>
    <loc>https://myblog.com/</loc>
    <lastmod>%s</lastmod>
    <changefreq>daily</changefreq>
    <priority>1.0</priority>
  </url>`, time.Now().Format("2006-01-02"))
    
    // 文章页面
    for _, article := range articles {
        sitemap += fmt.Sprintf(`
  <url>
    <loc>https://myblog.com/article/%s</loc>
    <lastmod>%s</lastmod>
    <changefreq>weekly</changefreq>
    <priority>0.8</priority>
  </url>`, article.Slug, article.UpdatedAt.Format("2006-01-02"))
    }
    
    sitemap += "\n</urlset>"
    
    c.Header("Content-Type", "application/xml")
    c.String(200, sitemap)
}
```

```go
// internal/service/article_service.go
func calculateReadingTime(content string) int {
    // 中文：500字/分钟，英文：200词/分钟
    chineseCount := len([]rune(content))
    englishCount := len(strings.Fields(content))
    
    minutes := (chineseCount / 500) + (englishCount / 200)
    if minutes < 1 {
        return 1
    }
    return minutes
}
```

#### 4.1 Redis 缓存机制

**目标**：减少数据库查询，提升响应速度

**需要缓存的数据**：
- 文章列表（缓存 5 分钟）
- 文章详情（缓存 10 分钟）
- 热门文章（缓存 30 分钟）
- 分类标签列表（缓存 1 小时）

**实现示例**：
```go
// internal/service/article_service.go
func (s *articleService) GetArticleDetail(slug string, clientIP string) (*response.ArticleDetailResponse, error) {
    // 尝试从缓存获取
    cacheKey := fmt.Sprintf("article:detail:%s", slug)
    if cached, err := s.cache.Get(cacheKey); err == nil {
        var article response.ArticleDetailResponse
        if json.Unmarshal([]byte(cached), &article) == nil {
            return &article, nil
        }
    }
    
    // 从数据库查询
    article, err := s.articleRepo.FindBySlug(slug)
    if err != nil {
        return nil, err
    }
    
    // 转换为响应格式
    result := &response.ArticleDetailResponse{ /* ... */ }
    
    // 缓存结果（10分钟）
    if data, _ := json.Marshal(result); len(data) > 0 {
        s.cache.Set(cacheKey, string(data), 10*time.Minute)
    }
    
    return result, nil
}
```

```vue
<script setup>
import { ref, onMounted } from 'vue'

const imageRef = ref(null)
const imageSrc = ref('')

onMounted(() => {
  const observer = new IntersectionObserver((entries) => {
    if (entries[0].isIntersecting) {
      imageSrc.value = props.src
      observer.disconnect()
    }
  })
  
  if (imageRef.value) {
    observer.observe(imageRef.value)
  }
})
</script>
```

```go
// 基于 ID 的游标分页
func (r *articleRepository) ListByCursor(cursor uint, limit int) ([]entity.Article, error) {
    var articles []entity.Article
    query := r.db.Where("status = ?", entity.ArticleStatusPublished)
    
    if cursor > 0 {
        query = query.Where("id < ?", cursor)

    err := query.
```

## 🆘 常见问题解决

### Q1: 如何判断一个功能是否需要优化？

**判断标准**：
1. 用户体验是否良好（加载快、提示清晰）
2. 功能是否完整（无明显缺陷）
3. 代码是否健壮（有错误处理）
4. 性能是否达标（响应时间、资源占用）

**优先优化**：影响用户核心操作的功能

---

### Q2: 前端优化和后端优化应该先做哪个？

**建议顺序**：
1. 先做前端 Loading 和错误处理（见效快，用户感知明显）
2. 再做后端缓存和索引优化（提升性能）
3. 最后做测试和部署配置（保证质量）

**原因**：用户体验的改善最直观，能快速看到效果，增强开发信心

---

### Q3: Redis 是必须的吗？

**回答**：不是必须，但强烈建议

**不使用 Redis 的情况**：
- 访问量很小（< 1000 PV/天）
- 服务器资源有限
- 只是个人博客学习项目

**使用 Redis 的优势**：
- 大幅降低数据库压力
- 提升响应速度（10倍以上）
- 支持更高并发量
- 实现更复杂的功能（限流、排行榜等）

---

### Q4: 测试覆盖率要达到多少？

**建议覆盖率**：
- **核心业务逻辑**：80%+（登录、文章 CRUD、评论）
- **工具函数**：90%+（JWT、密码加密、格式化）
- **Controller 层**：60%+（主要路由和中间件）
- **总体覆盖率**：60%+

**重点测试**：
1. 用户认证相关（安全性）
2. 数据操作相关（正确性）
3. 业务逻辑复杂的部分（可靠性）

---

### Q5: 如何平衡功能开发和代码质量？

**原则**：
- **快速迭代期**：功能优先，保证核心逻辑正确
- **稳定期**：质量优先，重构和优化代码
- **上线前**：必须通过基本测试和代码审查

**实践建议**：
1. 每开发一个功能，立即编写对应测试
2. 每周固定时间进行代码 review
3. 重要功能上线前必须有测试覆盖
4. 使用 Git 分支管理，主分支保持稳定
